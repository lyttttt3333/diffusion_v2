box_list = self.detect("box")

# Extract x coordinate
box_x = box_centroid_numpy[:, 0] # use x to determine right or left

# the box with the max x value is on the right column
tgt_box_idx = np.argmax(box_x)
tgt_box_pts = box_list[tgt_box_idx]
